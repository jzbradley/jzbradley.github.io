Rust Cheatsheet

Iterators

map

: Transforms each element of an iterator using a function/closure.

Example

let numbers = vec![1, 2, 3, 4, 5];
let squared: Vec<_> = numbers.iter().map(|&x| x * x).collect();
println!("{:?}", squared);

Output

[1, 4, 9, 16, 25]

filter

: Filters the elements of an iterator based on a predicate.

Example

let numbers = vec![1, 2, 3, 4, 5];
let evens: Vec<_> = numbers.iter().filter(|&&x| x % 2 == 0).collect();
println!("{:?}", evens);

Output

[2, 4]

fold

: (equivalent to `reduce` in some languages):** Accumulates the elements of an iterator.

Example

let numbers = vec![1, 2, 3, 4, 5];
let sum = numbers.iter().fold(0, |acc, &x| acc + x);
println!("{}", sum);

Output

15

all

: Checks if all elements of an iterator satisfy a predicate.

Example

let numbers = vec![2, 4, 6, 8, 10];
let are_all_even = numbers.iter().all(|&x| x % 2 == 0);
println!("{}", are_all_even);

Output

true

any

: Checks if any element of an iterator satisfies a predicate.

Example

let numbers = vec![1, 2, 3, 4, 5];
let has_even = numbers.iter().any(|&x| x % 2 == 0);
println!("{}", has_even);

Output

true

find

: Finds the first element satisfying a predicate.

Example

let numbers = vec![1, 2, 3, 4, 5];
let first_even = numbers.iter().find(|&&x| x % 2 == 0);
println!("{:?}", first_even);

Output

Some(&2)

collect

: Collects the elements of an iterator into a collection.

Example

let numbers = vec![1, 2, 3, 4, 5];
let evens: Vec<_> = numbers.iter().filter(|&&x| x % 2 == 0).collect();
println!("{:?}", evens);

Output

[2, 4]

flat_map

: Maps each element to an iterator and flattens the result into a single iterator.

Example

let numbers = vec![vec![1, 2], vec![3, 4], vec![5, 6]];
let flat: Vec<_> = numbers.iter().flat_map(|x| x.iter()).collect();
println!("{:?}", flat);

Output

[1, 2, 3, 4, 5, 6]
